        //【クラス】ポケモンのステータス
        function Status(number, name, agility) {
            this.number = number;
            this.name = name;
            this.agility = agility;
        }

        //【クラス】表示用ポケモンステータスStatus,すばやさ実数値,努力値,性格補正,個体値
        function PokemonData(status, effort, temper, individual) {
            this.status = status;
            this.effort = effort;
            this.temper = temper;
            this.individual = individual;
            

            //すばやさ実数値ここで計算
            this.agilityNum = ((status.agility * 2 + individual + effort / 4) * 50 / 100 + 5) * temper;
            
        }


        //ポケモンのステータスリストを作る
